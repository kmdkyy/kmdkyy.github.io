设置命令权限

  1. 安装文件访问控制列表
apt-get install acl
  1. 设置IP、DNS、主机名配置文件权限
setfacl -m u:bms:rw /etc/network/interfaces
setfacl -m u:bms:rw /etc/resolv.conf
setfacl -m u:bms:rw /etc/hosts
  1. 设置命令权限
cd /usr/bin/
chmod +s date
chmod +s hostnamectl
cd /sbin/
chmod +s shutdown
chmod +s reboot
